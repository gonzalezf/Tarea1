	CREATE OR REPLACE VIEW view_VentasVendedor AS
	SELECT v.ID_VENTA as ID_VENTA, c.NOMBRE as CLIENTE, v.MONTO_TOTAL as MONTO_TOTAL, v.FECHA as FECHA, v.TIME as HORA
	FROM VENTA v,
	(SELECT cl.RUT, cl.NOMBRE FROM CLIENTE cl) c,
	(SELECT us.RUT, us.NOMBRE FROM USUARIO us) u 
	WHERE v.ID_USUARIO = '1500' AND v.ID_CLIENTE = c.RUT AND v.ID_USUARIO = u.RUT
