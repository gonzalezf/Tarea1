CREATE OR REPLACE VIEW view_VentasCliente AS
	SELECT v.ID_VENTA as ID_VENTA, u.NOMBRE as VENDEDOR, v.MONTO_TOTAL as MONTO_TOTAL, v.FECHA as FECHA, v.TIME as HORA
	FROM VENTA v,
	(SELECT cl.RUT, cl.NOMBRE FROM CLIENTE cl) c,
	(SELECT us.RUT, us.NOMBRE FROM USUARIO us) u 
	WHERE v.ID_CLIENTE = '1500-2' AND v.ID_CLIENTE = c.RUT AND v.ID_USUARIO = u.RUT;
